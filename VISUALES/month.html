    qi=qi+1
    qj=0
    for j in v1:
        qj=qj+1
        if qj==qi+1:
            if j-i < 0:
                pn.append(i)
                pn.append(j)
            elif j-i > 0:
                pp.append(i)
                pp.append(j)

        if i==j:

def connectpoints(a,b,p1,p2):
    a1, a2 = a[p1], a[p2]
    b1, b2 = b[p1], b[p2]
    plt.plot([a1,a2],[b1,b2],'r-')

for i in range(0,len(a),2):
    connectpoints(a,b,i,i+1)
